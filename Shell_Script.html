Cheatsheet Of Shell Scripting!
  1. Steps to write and execute a script: 
    1. Open the terminal. Go to the directory where you want to create your script.
2. Create a file with .sh extension.
3. Write the script in the file using an editor.
4. Make the script executable with command chmod +x <fileName>.
5. Run the script using ./<fileName>.
  2. Comments: 
    #This is comment.
  3. Print: 
    name = "Hello World!!!"
echo $name
  4. variable: No need to specify the type of variable. 
    var1 = 23
var2 = 35.45
var3 = "sonu mahajan"
  5. Comparison Expressions: 
    >   -gt
<   -lt 
>=  -ge 
<=  -le 
==  -eq 
!=  -ne
  6. Input: 
    read x
echo $x
  7. if else Statement: 
    echo "Eneter any number: "
read no
if [ $no -gt 100 ]
then
echo "$no is greater then 100"
else
echo "$no is less then 100"
fi
  8. Nested if else Statement: 
    echo "Eneter any number: "
read no
if [ $no -gt 100 ]
then
        echo "$no is greater then 100"
elif [ $no == 50 ]
then
        echo "$no is half of 100"
else
        echo "$no is less then 100"
fi
  9. Case Statement: 
    read a
case $a in
        10) echo "Ten";;
        20) echo "Twenty";;
        30) echo "Thirty";;
        *) echo "Random number";;
esac
  10. While loop: 
    x=1
while [ $x -le 10 ]
do 
echo $x
x=`expr $x + 1`
done
  11. Delay Output: 
    x=1
while [ $x -le 10 ]
do 
echo $x
x=`expr $x + 1`
sleep 1
done
  12. For Loop(1): 
    for((x=1;x<=10;x++))
do 
echo $x

  13. For Loop(2):

  14. For Loop(3):

  15. Until Loop: 
    x=1
until [ $x -lt 10]
do 
echo $x
x=`expr $x + 1`
done

case $usr_cmd in
1)	echo "===================================="
    echo "ADD NEW CONTACT"
    echo "===================================="
    echo ""
    read -p "Enter Name: " name
    echo "Got that!"
    read -p "Enter Number: " number
        re='^[0-9]+$'
        if ! [[ $number =~ $re ]]
                then
            echo "error: Not a number"
        else
            clear
            echo "New Contact Info:-"
            echo ""
            echo "-> Name: $name"
            echo "-> Number: $number"
            echo "$name : $number" >> phonedir.log
            echo "Save Successfully"
        fi
;;
2)	echo "===================================="
    echo "SEARCH CONTACT"
    echo "===================================="
    echo ""
    read -p "Enter contact name to Search: " search_query
    clear
    echo "SEARCH RESULTS:"
    echo ""
    grep -i $search_query phonedir.log
;;
3)	echo "===================================="
    echo "DELETE CONTACT"
    echo "===================================="
    echo ""
    read -p "Enter contact name to Delete(case-sensitive): " delete_query
    sed -i -e "/$delete_query/d" phonedir.log
    echo "Delete Successful"

4)	echo "===================================="
    echo "PHONE DIRECTORY"
    echo "===================================="
    echo ""
    cat phonedir.log
;;

*)echo "INVALID OPTION";;

echo ""
read -p "Press 5 to Quit, Anything else to return Main Menu " confirm_exit
if [ $confirm_exit -eq 5 ]
then break
fi
done
